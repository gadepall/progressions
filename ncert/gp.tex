	\\
	\solution If the roots are $a, b$,

\item If $f$ is a function satisfying $$f\brak{x +y} = f\brak{x} f\brak{y} \forall x,y \in N$$ such that

	\\
	\solution Since 
\begin{align}
	f\brak{2}=
	f\brak{1}
	f\brak{1}
	=
	\sbrak{f\brak{1}}^2,
\end{align}
it is easy to verify that
\begin{align}
	f\brak{k}&=
	\sbrak{f\brak{1}}^k
	\\
	\therefore
\sum_{k = 1}^n f\brak{k} 
&=
\sum_{k = 1}^n 3^{k} 
	\\
	\frac{3\brak{3^{n}-1}}{2} &= 120,
	\\
	\text{or, }
	n =\frac{\log 81}{\log 3} &= 4
\end{align}
